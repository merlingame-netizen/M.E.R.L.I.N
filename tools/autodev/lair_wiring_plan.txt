## 4. PATTERN DISPOSE / CLEANUP CORRECT

`LairResult.dispose()` (lignes 775-794 de MerlinLairScene.ts) fait déjà :
  - window.removeEventListener('resize', onResize)
  - renderer.domElement.removeEventListener('mousemove', onMouseMove)
  - renderer.domElement.removeEventListener('click', onClick)
  - scene.traverse → geometry.dispose() + material.dispose() sur tous meshes/points
  - renderer.dispose()
  - removeChild(renderer.domElement) du DOM

Donc dans runMerlinLair :
  1. Annuler le RAF *avant* d'appeler dispose() pour éviter un render
     orphelin sur un renderer déjà libéré.
  2. Masquer et réinitialiser le wrapper div après dispose().
  3. Ne pas appeler dispose() depuis l'intérieur du RAF tick — toujours
     depuis le flux principal après cancelAnimationFrame.

// --- Lair Hub ---
async function runMerlinLair(app: HTMLElement): Promise<void> {
  // Retrieve or create the lair wrapper div
  let wrapper = document.getElementById('lair-canvas-wrapper') as HTMLDivElement | null;
  if (!wrapper) {
    wrapper = document.createElement('div');
    wrapper.id = 'lair-canvas-wrapper';
    wrapper.style.cssText = 'position:fixed;inset:0;width:100%;height:100%;z-index:10;';
    app.parentElement?.appendChild(wrapper) ?? document.body.appendChild(wrapper);
  }
  wrapper.style.display = 'block';

  // Fade in from black (menu already cut to black via cutToBlack())
  revealFromBlack(800);

  const lair = initMerlinLair(wrapper);

  // Real-time clock → day/night cycle (optional cosmetic, non-blocking)
  const now = new Date();
  lair.setTime({
    hour: now.getHours() + now.getMinutes() / 60,
    season: (['winter', 'spring', 'summer', 'autumn'] as const)[
      Math.floor(((now.getMonth() + 1) % 12) / 3)
    ] ?? 'spring',
  });

  // RAF loop
  let rafId = 0;
  let lastTs = performance.now();
  const tick = (): void => {
    rafId = requestAnimationFrame(tick);
    const now = performance.now();
    const dt = Math.min((now - lastTs) / 1000, 0.05);
    lastTs = now;
    lair.update(dt);
  };
  tick();

  // Wait for player to choose a zone
  const zone = await new Promise<import('./scenes/MerlinLairScene').LairZone>((resolve) => {
    lair.onZoneClick((z) => {
      // door  → start run immediately
      // others → log intent, stay in Lair (future overlays)
      if (z === 'door') {
        resolve(z);
        return;
      }
      // Future: dispatch overlay by zone
      // For now, log and keep Lair active
      console.info(`[Lair] zone clicked: ${z} (overlay not yet wired)`);
    });
  });

  // Cleanup: stop RAF first, then dispose renderer + event listeners
  cancelAnimationFrame(rafId);
  cutToBlack();
  await new Promise<void>((res) => setTimeout(res, 300)); // brief black frame
  lair.dispose();
  wrapper.style.display = 'none';

  void zone; // consumed — currently only 'door' resolves
}
